import java.awt.Color;
import java.awt.Dimension;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

import javax.swing.*;


public class Main {
    public static void main(String[] args) {
        {

            JFrame frame = new JFrame("");
            frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
            frame.setSize(800, 600);
            frame.setResizable(true);

            JPanel mainPanel = new JPanel();
            mainPanel.setPreferredSize(new Dimension(1500, 1000));

            JPanel subPanel1 = new JPanel();
            subPanel1.setPreferredSize(new Dimension(550, 250));
            subPanel1.add(new JLabel("tell us your mode, and oh here some motivation"));

            JPanel subPanel2 = new JPanel();
            subPanel2.setPreferredSize(new Dimension(400, 100));
            JRadioButton badbutton = new JRadioButton("Bad");
            JRadioButton okbutton = new JRadioButton("ok");
            JRadioButton AMAZINGbutton = new JRadioButton("AMAZING");

            ButtonGroup group = new ButtonGroup();
            group.add(badbutton);
            group.add(okbutton);
            group.add(AMAZINGbutton);

            badbutton.setBounds(120,300,70,40);
            okbutton.setBounds(220,300,70,40);
            AMAZINGbutton.setBounds(320,300,100,40);

            String[] names = { "Keep pushing forward!", "Embrace the grind.", "Chase your dreams.", "Rise and shine.",
                    "Stay the course", "Believe and achieve."};


            JButton button = new JButton("see your motivation?");

            button.addActionListener(new ActionListener() {
                public void actionPerformed(ActionEvent e) {
                    String name = names[(int) (Math.random() * names.length)];
                    ((JButton) e.getSource()).setText(name);
                }
            });

            frame.add(badbutton);
            mainPanel.add(button);
            mainPanel.add(subPanel1);
            mainPanel.add(subPanel2);
            frame.add(okbutton);
            frame.add(AMAZINGbutton);


            frame.getContentPane().add(mainPanel);
            frame.pack();
            frame.setVisible(true);
        }

        }
}
